CREATE TABLE locatie (
  id_locatie   NUMBER GENERATED BY DEFAULT AS IDENTITY,
  adresa       VARCHAR2(200) NOT NULL,
  oras         VARCHAR2(100) NOT NULL,
  judet        VARCHAR2(100) NOT NULL,
  CONSTRAINT pk_locatie PRIMARY KEY (id_locatie)
);

CREATE TABLE campanie (
  id_campanie  NUMBER GENERATED BY DEFAULT AS IDENTITY,
  nume_campanie VARCHAR2(150) NOT NULL,
  descriere     VARCHAR2(1000),
  data_start    DATE NOT NULL,
  data_final    DATE,
  status        VARCHAR2(30) NOT NULL,
  CONSTRAINT pk_campanie PRIMARY KEY (id_campanie),
  CONSTRAINT ck_campanie_status CHECK (status IN ('PLANIFICATA','ACTIVA','INCHISA','ANULATA')),
  CONSTRAINT ck_campanie_date CHECK (data_final IS NULL OR data_final >= data_start)
);

CREATE TABLE organizatie (
  id_organizatie   NUMBER GENERATED BY DEFAULT AS IDENTITY,
  nume_organizatie VARCHAR2(150) NOT NULL,
  email            VARCHAR2(150),
  telefon          VARCHAR2(30),
  website          VARCHAR2(200),
  CONSTRAINT pk_organizatie PRIMARY KEY (id_organizatie),
  CONSTRAINT uq_organizatie_email UNIQUE (email)
);

CREATE TABLE sponsor (
  id_sponsor   NUMBER GENERATED BY DEFAULT AS IDENTITY,
  nume_sponsor VARCHAR2(150) NOT NULL,
  tip_sponsor  VARCHAR2(30) NOT NULL,
  email        VARCHAR2(150),
  telefon      VARCHAR2(30),
  CONSTRAINT pk_sponsor PRIMARY KEY (id_sponsor),
  CONSTRAINT ck_sponsor_tip CHECK (tip_sponsor IN ('PERSOANA','COMPANIE','ONG','ALTELE')),
  CONSTRAINT uq_sponsor_email UNIQUE (email)
);

CREATE TABLE categorie_voluntar (
  id_categorie_voluntar NUMBER GENERATED BY DEFAULT AS IDENTITY,
  nume_categorie        VARCHAR2(50) NOT NULL,
  descriere             VARCHAR2(1000),
  CONSTRAINT pk_categorie_voluntar PRIMARY KEY (id_categorie_voluntar),
  CONSTRAINT ck_cat_vol_nume CHECK (nume_categorie IN ('INCEPATOR','MEDIU','AVANSAT','COORDONATOR'))
);

CREATE TABLE competenta (
  id_competenta   NUMBER GENERATED BY DEFAULT AS IDENTITY,
  nume_competenta VARCHAR2(120) NOT NULL,
  CONSTRAINT pk_competenta PRIMARY KEY (id_competenta),
  CONSTRAINT uq_competenta_nume UNIQUE (nume_competenta)
);

CREATE TABLE voluntar (
  id_voluntar          NUMBER GENERATED BY DEFAULT AS IDENTITY,
  id_categorie_voluntar NUMBER NOT NULL,
  nume                VARCHAR2(80) NOT NULL,
  prenume             VARCHAR2(80) NOT NULL,
  email               VARCHAR2(150),
  telefon             VARCHAR2(30),
  data_nasterii       DATE,
  CONSTRAINT pk_voluntar PRIMARY KEY (id_voluntar),
  CONSTRAINT fk_voluntar_categorie
    FOREIGN KEY (id_categorie_voluntar) REFERENCES categorie_voluntar(id_categorie_voluntar),
  CONSTRAINT uq_voluntar_email UNIQUE (email)
);

CREATE TABLE proiect (
  id_proiect     NUMBER GENERATED BY DEFAULT AS IDENTITY,
  id_organizatie NUMBER NOT NULL,
  id_campanie    NUMBER,
  id_locatie     NUMBER NOT NULL,
  nume_proiect   VARCHAR2(150) NOT NULL,
  descriere      VARCHAR2(1000),
  data_start     DATE NOT NULL,
  data_final     DATE,
  status         VARCHAR2(30) NOT NULL,
  CONSTRAINT pk_proiect PRIMARY KEY (id_proiect),
  CONSTRAINT fk_proiect_organizatie
    FOREIGN KEY (id_organizatie) REFERENCES organizatie(id_organizatie),
  CONSTRAINT fk_proiect_campanie
    FOREIGN KEY (id_campanie) REFERENCES campanie(id_campanie),
  CONSTRAINT fk_proiect_locatie
    FOREIGN KEY (id_locatie) REFERENCES locatie(id_locatie),
  CONSTRAINT ck_proiect_status CHECK (status IN ('IN_DESFASURARE','FINALIZAT','ANULAT')),
  CONSTRAINT ck_proiect_date CHECK (data_final IS NULL OR data_final >= data_start)
);

CREATE TABLE voluntar_competenta (
  id_voluntar   NUMBER NOT NULL,
  id_competenta NUMBER NOT NULL,
  nivel         VARCHAR2(30) NOT NULL,
  CONSTRAINT pk_vol_comp PRIMARY KEY (id_voluntar, id_competenta),
  CONSTRAINT fk_vc_voluntar FOREIGN KEY (id_voluntar) REFERENCES voluntar(id_voluntar),
  CONSTRAINT fk_vc_competenta FOREIGN KEY (id_competenta) REFERENCES competenta(id_competenta),
  CONSTRAINT ck_vc_nivel CHECK (nivel IN ('INCEPATOR','MEDIU','AVANSAT','EXPERT'))
);

CREATE TABLE proiect_competenta (
  id_proiect    NUMBER NOT NULL,
  id_competenta NUMBER NOT NULL,
  CONSTRAINT pk_proj_comp PRIMARY KEY (id_proiect, id_competenta),
  CONSTRAINT fk_pc_proiect FOREIGN KEY (id_proiect) REFERENCES proiect(id_proiect),
  CONSTRAINT fk_pc_competenta FOREIGN KEY (id_competenta) REFERENCES competenta(id_competenta)
);

CREATE TABLE proiect_sponsor (
  id_proiect NUMBER NOT NULL,
  id_sponsor NUMBER NOT NULL,
  CONSTRAINT pk_proj_sponsor PRIMARY KEY (id_proiect, id_sponsor),
  CONSTRAINT fk_ps_proiect FOREIGN KEY (id_proiect) REFERENCES proiect(id_proiect),
  CONSTRAINT fk_ps_sponsor FOREIGN KEY (id_sponsor) REFERENCES sponsor(id_sponsor)
);

CREATE TABLE participare_voluntar_proiect (
  id_voluntar NUMBER NOT NULL,
  id_proiect  NUMBER NOT NULL,
  rol         VARCHAR2(80) NOT NULL,
  numar_ore   NUMBER(7,2) DEFAULT 0 NOT NULL,
  CONSTRAINT pk_participare PRIMARY KEY (id_voluntar, id_proiect),
  CONSTRAINT fk_pvp_voluntar FOREIGN KEY (id_voluntar) REFERENCES voluntar(id_voluntar),
  CONSTRAINT fk_pvp_proiect FOREIGN KEY (id_proiect) REFERENCES proiect(id_proiect),
  CONSTRAINT ck_pvp_ore CHECK (numar_ore >= 0)
);

CREATE TABLE donatie (
  id_donatie  NUMBER GENERATED BY DEFAULT AS IDENTITY,
  id_proiect  NUMBER NOT NULL,
  data_donatie DATE NOT NULL,
  donator     VARCHAR2(150),
  suma        NUMBER(12,2) NOT NULL,
  CONSTRAINT pk_donatie PRIMARY KEY (id_donatie),
  CONSTRAINT fk_donatie_proiect FOREIGN KEY (id_proiect) REFERENCES proiect(id_proiect),
  CONSTRAINT ck_donatie_suma CHECK (suma > 0)
);

-- script inserare voluntar_competenta

SET SERVEROUTPUT ON;

DECLARE
  v_min_vol NUMBER := 1;
  v_max_vol NUMBER := 120;

  v_c1 NUMBER;
  v_c2 NUMBER;
  v_c3 NUMBER;

  v_lvl1 VARCHAR2(30);
  v_lvl2 VARCHAR2(30);
  v_lvl3 VARCHAR2(30);

  v_inserted NUMBER := 0;

  -- functie: nivel competenta
  FUNCTION pick_level(p_vol_id NUMBER, p_slot NUMBER) RETURN VARCHAR2 IS
  BEGIN
    IF p_vol_id >= 116 THEN
      IF p_slot = 2 THEN RETURN 'AVANSAT'; ELSE RETURN 'EXPERT'; END IF;
    ELSIF p_vol_id >= 101 THEN
      IF p_slot = 1 THEN RETURN 'AVANSAT'; ELSE RETURN 'EXPERT'; END IF;
    ELSIF p_vol_id >= 61 THEN
      IF p_slot = 1 THEN RETURN 'MEDIU'; ELSE RETURN 'AVANSAT'; END IF;
    ELSE
      IF p_slot = 1 THEN RETURN 'INCEPATOR';
      ELSIF p_slot = 2 THEN RETURN 'MEDIU';
      ELSE RETURN 'INCEPATOR';
      END IF;
    END IF;
  END;

  -- functie: competentă determinista (1..20)
  FUNCTION comp_id(p_vol_id NUMBER, p_slot NUMBER) RETURN NUMBER IS
  BEGIN
    IF p_slot = 1 THEN
      RETURN MOD(p_vol_id - 1, 20) + 1;
    ELSIF p_slot = 2 THEN
      RETURN MOD(p_vol_id + 4, 20) + 1;
    ELSE
      RETURN MOD(p_vol_id + 9, 20) + 1;
    END IF;
  END;

BEGIN
  FOR v IN v_min_vol .. v_max_vol LOOP
    v_c1 := comp_id(v, 1);
    v_c2 := comp_id(v, 2);
    v_c3 := comp_id(v, 3);

    -- asigură 3 competente diferite
    IF v_c2 = v_c1 THEN v_c2 := MOD(v_c2, 20) + 1; END IF;

    IF v_c3 IN (v_c1, v_c2) THEN v_c3 := MOD(v_c3 + 1, 20) + 1; END IF;
    IF v_c3 IN (v_c1, v_c2) THEN v_c3 := MOD(v_c3 + 1, 20) + 1; END IF;

    -- calculeaza nivelurile in variabile PL/SQL
    v_lvl1 := pick_level(v, 1);
    v_lvl2 := pick_level(v, 2);
    v_lvl3 := pick_level(v, 3);

    INSERT INTO voluntar_competenta (id_voluntar, id_competenta, nivel)
    VALUES (v, v_c1, v_lvl1);

    INSERT INTO voluntar_competenta (id_voluntar, id_competenta, nivel)
    VALUES (v, v_c2, v_lvl2);

    INSERT INTO voluntar_competenta (id_voluntar, id_competenta, nivel)
    VALUES (v, v_c3, v_lvl3);

    v_inserted := v_inserted + 3;
  END LOOP;

  COMMIT;
  DBMS_OUTPUT.PUT_LINE('Inserari efectuate: ' || v_inserted);
END;
/

COMMIT;

-- insert participare_voluntar_proiect


DECLARE
  v_proiect_id   NUMBER;
  v_rol          VARCHAR2(80);
  v_ore          NUMBER(7,2);
BEGIN
  -- Pentru fiecare voluntar
  FOR v IN (SELECT id_voluntar FROM voluntar) LOOP

    -- Fiecare voluntar participa la 2 proiecte
    FOR i IN 1..2 LOOP

      -- Alegem un proiect aleator (ID 2–22)
      v_proiect_id := TRUNC(DBMS_RANDOM.VALUE(2, 23));

      -- Alegem rolul
      CASE TRUNC(DBMS_RANDOM.VALUE(1, 5))
        WHEN 1 THEN v_rol := 'Voluntar';
        WHEN 2 THEN v_rol := 'Coordonator Activitati';
        WHEN 3 THEN v_rol := 'Responsabil Logistica';
        WHEN 4 THEN v_rol := 'Asistent Proiect';
      END CASE;

      -- Numar de ore realist
      v_ore := TRUNC(DBMS_RANDOM.VALUE(10, 120), 2);

      BEGIN
        INSERT INTO participare_voluntar_proiect
          (id_voluntar, id_proiect, rol, numar_ore)
        VALUES
          (v.id_voluntar, v_proiect_id, v_rol, v_ore);

      EXCEPTION
        WHEN DUP_VAL_ON_INDEX THEN
          NULL; -- evitam dublurile (PK)
      END;

    END LOOP;
  END LOOP;

  COMMIT;
END;
/

DECLARE
  v_data   DATE;
  v_suma   NUMBER(12,2);
  v_donator VARCHAR2(150);
BEGIN
  -- Pentru fiecare proiect
  FOR p IN (SELECT id_proiect FROM proiect) LOOP

    -- Fiecare proiect primeste intre 10 si 30 donatii
    FOR i IN 1 .. TRUNC(DBMS_RANDOM.VALUE(10, 31)) LOOP

      -- Data donatie (2023–2025)
      v_data := DATE '2023-01-01'
                + TRUNC(DBMS_RANDOM.VALUE(0, 1095));

      -- Suma donata
      v_suma := ROUND(DBMS_RANDOM.VALUE(50, 20000), 2);

      -- Donator
      CASE TRUNC(DBMS_RANDOM.VALUE(1, 6))
        WHEN 1 THEN v_donator := 'Persoana fizica';
        WHEN 2 THEN v_donator := 'Companie privata';
        WHEN 3 THEN v_donator := 'Fundatie';
        WHEN 4 THEN v_donator := 'Sponsor local';
        WHEN 5 THEN v_donator := 'Donator anonim';
      END CASE;

      INSERT INTO donatie
        (id_proiect, data_donatie, donator, suma)
      VALUES
        (p.id_proiect, v_data, v_donator, v_suma);

    END LOOP;
  END LOOP;

  COMMIT;
END;
/

COMMIT;
